110 //Use OpenGL 3.3 core
111 SDL_GL_SetAttribute( SDL_GL_CONTEXT_MAJOR_VERSION, 3 );

125
126 // Check if Window did not create.

128 errorStream << "Window could not be created! SDL Error: " << SDL_GetError() << "\n";
129 success = false;
130 }
131
132 //Create an OpenGL Graphics Context
133 gContext = SDL_GL_CreateContext( gWindow );

147 errorStream << "Unable to initialize OpenGL!\n";
148 success = false;

150 }
151
