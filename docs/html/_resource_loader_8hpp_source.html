117void ResourceLoader<ResourceT>::Destroy(std::shared_ptr<ResourceT>& resource) {
118 if (resource.use_count() > 2) {
119 resource.reset();
120 return;
121 }
122
123 if (!resource) {
124 // Pointer was already freed
125 return;
126 }
127
128 auto resourceIt = FindByPtr(resource);
129
130 if (resourceIt == m_resources.end()) {
131 // Resource has already been freed
132 return;

134
135 resource.reset();
136 // There are no more references to the resource; Free it
137
138 // There is no need to call this because the deleter passed to the shared_ptr will call it
139 // when the last reference is destroyed (it is removed from the map).
140
141 // WARN: Will cause memory leak if you do not pass a deleter to the shared_ptr
142
143 m_resources.erase(resourceIt->first);
144}
